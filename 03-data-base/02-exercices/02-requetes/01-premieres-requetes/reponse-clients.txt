** 3.3. Selection de tous les clients habitants au CP = 4800

SELECT * 
FROM `clients` 
WHERE CP =4800



** 3.4. Selection de tous les clients habitants au CP = 4800 ou = 4020

SELECT * 
FROM `clients` 
WHERE CP =4800
OR CP =4020



** 3.5. Selection de tous les clients habitants la province de Liege : au CP > 4000 et < 4999

SELECT * 
FROM `clients` 
WHERE CP >4000
AND CP <4999



** 3.6. Selection de tous les clients avec Prenom = John

SELECT * 
FROM `clients` 
WHERE `Prenom` LIKE 'John'

ou

SELECT * 
FROM `clients` 
WHERE `Prenom` = 'John'



** 3.7. Selection de tous les clients avec Prenom = John ou CP = 4020

SELECT * 
FROM `clients` 
WHERE `Prenom` LIKE 'John'
OR `CP` =4020



** 3.8. Selection de tous les clients avec "BOU" dans le Nom

SELECT * 
FROM `clients` 
WHERE `Nom` LIKE '%BOU%'



** 3.9. Compter tous les clients 

SELECT count( * ) 
FROM `clients` 



** 3.10. Compter tous les clients avec "BOU" dans le Nom

SELECT count( * ) 
FROM `clients` 
WHERE `Nom` LIKE '%BOU%'



** 3.11. Selection de tous les clients tries par Nom, Prenom ascendant

SELECT * 
FROM `clients` 
ORDER BY Nom ASC , Prenom ASC 



** 3.12. Selection de tous les clients tries par Nom, Prenom descendant

SELECT * 
FROM `clients` 
ORDER BY Nom DESC , Prenom DESC 



** 3.13. Selection de tous les clients groupes par Nom

SELECT Nom
FROM Clients
GROUP BY NOM

ou 

SELECT DISTINCT Nom 
FROM Clients

ou 

SELECT DISTINCT Nom 
FROM Clients
ORDER BY Nom ASC



** 3.14. Selection de tous les clients groupes par Nom et comptes

SELECT NOM, count(*)
FROM Clients
GROUP BY NOM

ou 


SELECT NOM, count(*) AS Nombre
FROM Clients
GROUP BY NOM



** 3.15. Selection des clients groupes par localite et comptes

SELECT Localite, count( * ) 
FROM Clients

ou 

SELECT Localite, count(*) AS Nombre

** 3.16. Selection des clients groupes par categorie pour chaque localite et comptes
